    unsigned int silnia(unsigned int n)
    {
    	int s = 1; /* zwracana wartoœæ silni */
    	int i;
     
    	/* implementacja nierekurencyjna */
    	for(i = 2; i <= n; i++) 
    		s *= i;
    	return(s);
    }
