    int main(void)
    {
     
    	int A[N] = {-2,-1,-3};
    	int i;
     
    	i = rand();
     
    	if( (A[i] > 0) &&  (i < N)  )

    	}
    	else
    	{
    		printf("PUDLO\n");
    	}
     
    	return(0);
     
    }
